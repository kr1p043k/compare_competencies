def load_market_competencies() -> Dict[str, int]:
    """Загрузка рыночных компетенций из data/processed/competency_frequency_mapped.json."""
    filepath = PROCESSED_DIR / "competency_frequency_mapped.json"
    if filepath.exists():
        try:
            with open(filepath, 'r', encoding='utf-8') as f:
                return json.load(f)
        except Exception as e:
            logger.error(f"Ошибка загрузки market_competencies: {e}")
    return {}


def load_all_profiles() -> Dict[str, list]:
    """Загрузка профилей студентов из data/students/."""
    profiles = {}
    for name in ['base', 'dc', 'top_dc']:
        filepath = STUDENTS_DIR / f"{name}_competency.json"
        if filepath.exists():
            try:
                with open(filepath, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                    profiles[name] = data.get("навыки", [])
            except Exception as e:
                logger.error(f"Ошибка загрузки профиля {name}: {e}")
                profiles[name] = []
        else:
            profiles[name] = []
    return profiles


def load_competency_descriptions() -> Dict[str, str]:
    """Загрузка описаний компетенций из description_of_competency.txt."""
    descriptions = {}
    filepath = STUDENTS_DIR / "descriptiom_of_competency.txt"
    if filepath.exists():
        try:
            with open(filepath, 'r', encoding='utf-8') as f:
                lines = f.read().splitlines()
            for line in lines[1:]:
                line = line.strip()
                if '\t' in line:
                    parts = line.split('\t', 1)
                    if len(parts) == 2:
                        descriptions[parts[0]] = parts[1]
        except Exception as e:
            logger.error(f"Ошибка загрузки описаний: {e}")
    return descriptions

# 🚀 ЗАПУСК CLI-КОМАНД
# ============================================
async def run_command(cmd: list, timeout: int = 600, description: str = "") -> Dict[str, Any]:
    """
    Запуск команды через subprocess с принудительным UTF-8 для Windows.
    Возвращает {"success": bool, "stdout": str, "stderr": str, "returncode": int, "description": str}
    """
    logger.info(f"Запуск команды: {' '.join(cmd)}")
    
    try:
        # 🔧 Windows fix: устанавливаем PYTHONIOENCODING=utf-8 и PYTHONUTF8=1
        env = os.environ.copy()
        env['PYTHONIOENCODING'] = 'utf-8'
        env['PYTHONUTF8'] = '1'
        
        process = subprocess.Popen(
            cmd,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            cwd=str(PROJECT_ROOT),
            text=True,
            encoding='utf-8',
            errors='replace',
            env=env
        )
        stdout, stderr = process.communicate(timeout=timeout)
        
        success = process.returncode == 0
        if success:
            logger.info(f"Команда выполнена успешно: {description}")
        else:
            logger.error(f"Ошибка выполнения ({process.returncode}): {description}")
            if stderr:
                logger.error(f"Stderr: {stderr[-1000:]}")
        
        return {
            "success": success,
            "description": description,
            "stdout": stdout[-5000:] if stdout else "",
            "stderr": stderr[-2000:] if stderr else "",
            "returncode": process.returncode
        }
    except subprocess.TimeoutExpired:
        process.kill()
        logger.error(f"Таймаут команды: {description}")
        return {
            "success": False,
            "description": description,
            "stdout": "",
            "stderr": f"Превышено времени выполнения ({timeout}с)",
            "returncode": -1
        }
    except Exception as e:
        logger.exception(f"Исключение при запуске: {description}")
        return {
            "success": False,
            "description": description,
            "stdout": "",
            "stderr": str(e),
            "returncode": -1
        }

@app.get("/api/profiles")
async def get_profiles():
    """Получить все профили студентов."""
    profiles = load_all_profiles()
    market = load_market_competencies()
    
    result = {}
    for name, skills in profiles.items():
        common = set(skills) & set(market.keys())
        result[name] = {
            "skills_count": len(skills),
            "skills": sorted(skills),
            "common_with_market": len(common),
            "coverage_percent": round(len(common) / len(market) * 100, 1) if market else 0
        }
    
    return result


@app.get("/api/profiles/{name}")
async def get_profile_detail(name: str):
    """Детальная информация о профиле: метрики, gap, покрытие."""
    try:
        from src.analyzers.gap_analyzer import calculate_coverage_metrics, calculate_gap_analysis
    except ImportError as e:
        raise HTTPException(status_code=500, detail=f"Ошибка импорта модулей: {e}")
    
    market = load_market_competencies()
    profiles = load_all_profiles()
    descriptions = load_competency_descriptions()
    
    if name not in profiles:
        raise HTTPException(status_code=404, detail=f"Профиль '{name}' не найден. Доступны: {list(profiles.keys())}")
    
    skills = profiles[name]
    metrics = calculate_coverage_metrics(skills, market)
    gap = calculate_gap_analysis(skills, market)
    
    skills_with_desc = []
    for skill in sorted(skills):
        skills_with_desc.append({
            "code": skill,
            "description": descriptions.get(skill, ""),
            "market_frequency": market.get(skill, 0)
        })
    
    return {
        "profile": name,
        "metrics": {
            "program_size": metrics['program_size'],
            "market_size": metrics['market_size'],
            "common_count": metrics['common_count'],
            "coverage_percent": round(metrics['coverage'] * 100, 1),
            "weighted_coverage_percent": round(metrics['weighted_coverage'] * 100, 1),
            "coverage_details": metrics.get('coverage_details', {})
        },
        "gap": {
            "total_missing": gap['total_missing'],
            "high_demand": [
                {"skill": s, "frequency": f} for s, f in gap['high_demand']
            ],
            "medium_demand": [
                {"skill": s, "frequency": f} for s, f in gap['medium_demand']
            ],
            "low_demand": [
                {"skill": s, "frequency": f} for s, f in gap['low_demand']
            ]
        },
        "skills": skills_with_desc
    }

@app.post("/api/pipeline/step1-collect")
async def step1_collect_vacancies(
    query: str = Query("Data Scientist", description="Поисковый запрос"),
    area_id: int = Query(1, description="ID региона (1=Москва, 2=СПб, 1006=Россия)"),
    max_pages: int = Query(5, ge=1, le=20, description="Количество страниц (1-20)"),
    period: int = Query(30, ge=1, le=90, description="Период в днях"),
    it_sector: bool = Query(False, description="Поиск по всему IT-сектору"),
    excel: bool = Query(False, description="Сохранить в Excel")
):
    """
    Шаг 1: Сбор вакансий с hh.ru.
    Соответствует: python main.py --query "..." --area-id ... --max-pages ...
    """
    cmd = [sys.executable, str(PROJECT_ROOT / "main.py")]
    
    if it_sector:
        cmd.append("--it-sector")
    else:
        cmd.extend(["--query", query])
        cmd.extend(["--area-id", str(area_id)])
        cmd.extend(["--max-pages", str(max_pages)])
        cmd.extend(["--period", str(period)])
    
    if excel:
        cmd.append("--excel")
    
    cmd.extend(["--skip-details", "--no-filter"])
    
    return await run_command(cmd, description="Сбор вакансий")

@app.post("/api/pipeline/full-cycle")
async def full_cycle(
    query: str = Query("Data Scientist"),
    area_id: int = Query(1),
    max_pages: int = Query(5),
    period: int = Query(30),
    it_sector: bool = Query(False)
):
    """
    Полный цикл: Сбор → Кластеры → ML-модель → Gap-анализ.
    Соответствует полному README Quick Start.
    """
    steps = {}
    
    # Шаг 1: Сбор вакансий
    steps["step1_collect"] = await step1_collect_vacancies(
        query=query, area_id=area_id, max_pages=max_pages,
        period=period, it_sector=it_sector
    )
    if not steps["step1_collect"]["success"]:
        return {"status": "failed_at_step1", "steps": steps}
    
    # Шаг 2: Кластеры
    steps["step2_clusters"] = await step2_train_clusters("all")
    if not steps["step2_clusters"]["success"]:
        return {"status": "failed_at_step2", "steps": steps}
    
    # Шаг 3: ML-модель
    steps["step3_model"] = await step3_train_model()
    if not steps["step3_model"]["success"]:
        return {"status": "failed_at_step3", "steps": steps}
    
    # Шаг 4: Gap-анализ
    steps["step4_gap_analysis"] = await step4_gap_analysis()
    
    return {
        "status": "completed" if steps["step4_gap_analysis"]["success"] else "completed_with_errors",
        "steps": steps
    }

@app.post("/api/upload/profile")
async def upload_profile(file: UploadFile = File(...)):
    """Загрузить JSON-профиль студента."""
    if not file.filename.endswith('.json'):
        raise HTTPException(status_code=400, detail="Только JSON файлы")
    
    try:
        content = await file.read()
        data = json.loads(content.decode('utf-8'))
        
        if "навыки" not in data or not isinstance(data["навыки"], list):
            raise HTTPException(
                status_code=400,
                detail="JSON должен содержать ключ 'навыки' со списком компетенций"
            )
        
        filepath = STUDENTS_DIR / file.filename
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        
        logger.info(f"Загружен профиль: {file.filename} ({len(data['навыки'])} навыков)")
        
        return {
            "status": "ok",
            "filename": file.filename,
            "skills_count": len(data["навыки"])
        }
    except json.JSONDecodeError:
        raise HTTPException(status_code=400, detail="Некорректный JSON")
    except Exception as e:
        logger.exception("Ошибка загрузки профиля")
        raise HTTPException(status_code=500, detail=str(e))

    cmd = [sys.executable, str(PROJECT_ROOT / "scripts" / "extend_it_skills.py")]
    if auto_confirm:
        cmd.append("--yes")
    return await run_command(cmd, description="Расширение it_skills.json")
